List<Method> get allInstanceMethods {
  if (_allInstanceMethods != null) return _allInstanceMethods;
  _allInstanceMethods = []
    ..addAll([]
      ..addAll(instanceMethods)
      ..sort(byName))
    ..addAll([]
      ..addAll(inheritedMethods)
      ..sort(byName));
  return _allInstanceMethods;
}

bool get isParameterizedType {
  if (_type is FunctionType) {
    return typeArguments.isNotEmpty;
  } else if (_type is ParameterizedType) {
    return (_type as ParameterizedType).typeArguments.isNotEmpty;
  }
  return false;
}
